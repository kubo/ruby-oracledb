def initialize(key, val)
  @name = key
  @bitflag = val["bitflag"]
  prefix = val["prefix"]
  @to_dpi = @from_dpi = false
  case val["dir"]
  when "to_dpi"
    @to_dpi = true
  when "from_dpi"
    @from_dpi = true
  when "both"
    @to_dpi = true
    @from_dpi = true
  end
  @values = val["values"].map do |val|
    [prefix + val.upcase, val]
  end
end
